def model_func(x, A, K, C):
    return A * np.exp(K * x) + C

def fit_exp_linear(x, y, C=0):
    """fit an exponential"""
    y = y - C
    y = np.log(y)
    K, A_log = np.polyfit(x, y, 1)
    A = np.exp(A_log)
    return A, K
